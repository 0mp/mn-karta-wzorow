% Interpolacja Lagrange'a.
\entry
\textbf{I. L.}:
$w(x_i) = f(x_i)$;

% Interpolacja Hermite'a.
\entry
\textbf{I. Hermite'a}:
$\deriv{w}{k}(x_i) = \deriv{f}{k}_i, k = 0 \ldots m$;

\entry

to
